62 private:
63  std::vector<std::string> createVector(const char** events) {
64  std::vector<std::string> vec;
65  if ( events == NULL ) return vec;
66  const char** ev = events;
67  while ( NULL != *ev ) {
68  vec.push_back(*ev);
69  ev++;
70  }
71  return vec;
72  }
